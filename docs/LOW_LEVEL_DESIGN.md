## 2. Database Schema (`src/db/schema.sql`)

```sql
-- =========================================================================
-- PRIMARY TABLES
-- =========================================================================

CREATE TABLE IF NOT EXISTS customers (
    customer_id    INTEGER     PRIMARY KEY,
    email          TEXT        NOT NULL,
    full_name      TEXT        NOT NULL,
    signup_date    DATE        NOT NULL,
    country_code   CHAR(2),
    is_active      BOOLEAN     NOT NULL DEFAULT TRUE,
    CONSTRAINT customers_email_lowercase_chk CHECK (email = lower(email)),
    CONSTRAINT customers_email_unique UNIQUE (email)
);

CREATE TABLE IF NOT EXISTS orders (
    order_id       BIGINT                   PRIMARY KEY,
    customer_id    INTEGER                  NOT NULL REFERENCES customers(customer_id),
    order_ts       TIMESTAMP WITH TIME ZONE NOT NULL,
    status         TEXT                     NOT NULL,
    total_amount   NUMERIC(12,2)            NOT NULL,
    currency       CHAR(3)                  NOT NULL,
    CONSTRAINT orders_status_chk CHECK (status IN ('placed','shipped','cancelled','refunded'))
);

CREATE TABLE IF NOT EXISTS order_items (
    order_id     BIGINT         NOT NULL REFERENCES orders(order_id),
    line_no      INTEGER        NOT NULL,
    sku          TEXT           NOT NULL,
    quantity     INTEGER        NOT NULL,
    unit_price   NUMERIC(12,2)  NOT NULL,
    category     TEXT,
    PRIMARY KEY (order_id, line_no),
    CONSTRAINT order_items_qty_positive_chk   CHECK (quantity   > 0),
    CONSTRAINT order_items_price_positive_chk CHECK (unit_price > 0)
);

-- =========================================================================
-- QUARANTINE TABLES — same shape as source (all TEXT to preserve raw), plus
-- audit columns. No FKs by design — they hold rejects.
-- =========================================================================

CREATE TABLE IF NOT EXISTS quarantine_customers (
    customer_id        TEXT,
    email              TEXT,
    full_name          TEXT,
    signup_date        TEXT,
    country_code       TEXT,
    is_active          TEXT,
    quarantine_reason  TEXT        NOT NULL,
    source_file        TEXT        NOT NULL,
    ingested_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS quarantine_orders (
    order_id           TEXT,
    customer_id        TEXT,
    order_ts           TEXT,
    status             TEXT,
    total_amount       TEXT,
    currency           TEXT,
    quarantine_reason  TEXT        NOT NULL,
    source_file        TEXT        NOT NULL,
    ingested_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS quarantine_order_items (
    order_id           TEXT,
    line_no            TEXT,
    sku                TEXT,
    quantity           TEXT,
    unit_price         TEXT,
    category           TEXT,
    quarantine_reason  TEXT        NOT NULL,
    source_file        TEXT        NOT NULL,
    ingested_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- =========================================================================
-- INDEXES — lean, only where views need them
-- =========================================================================
CREATE INDEX IF NOT EXISTS idx_orders_customer_id ON orders(customer_id);
CREATE INDEX IF NOT EXISTS idx_orders_order_ts    ON orders(order_ts);
CREATE INDEX IF NOT EXISTS idx_order_items_sku    ON order_items(sku);
```

### Schema Decisions (Rationale)

| Decision | Rationale |
|---|---|
| `customers.email UNIQUE` on lowercase value | Brief: "enforce uniqueness on stored value, avoid extensions." Enforced at write time via `CHECK (email = lower(email))` + `UNIQUE`. |
| `orders.status` as `TEXT` + CHECK vs ENUM | ENUMs are harder to evolve (adding values needs ALTER TYPE). `CHECK` is portable and explicit. |
| `order_items.quantity > 0` + `unit_price > 0` as CHECKs | Brief requires positive values; enforcing at schema level = defense in depth. Python already filters, but DB is the last line. |
| No FKs on quarantine tables | Quarantine by definition violates referential integrity. |
| Quarantine columns as `TEXT` | Preserves *exactly* what was in the source file, including invalid formats. |

## 3. SQL Views (`src/db/views.sql`)

```sql
-- =========================================================================
-- ANALYTICS VIEWS
-- =========================================================================

-- Daily metrics: orders_count, total_revenue, AOV
CREATE OR REPLACE VIEW v_daily_metrics AS
SELECT
    (order_ts AT TIME ZONE 'UTC')::date  AS date,
    COUNT(*)                             AS orders_count,
    SUM(total_amount)                    AS total_revenue,
    ROUND(AVG(total_amount), 2)          AS average_order_value
FROM orders
GROUP BY 1
ORDER BY 1;

-- Top 10 customers by lifetime spend (uses window function for rank)
CREATE OR REPLACE VIEW v_top_customers AS
WITH ranked AS (
    SELECT
        c.customer_id,
        c.full_name,
        c.email,
        SUM(o.total_amount) AS lifetime_spend,
        COUNT(o.order_id)   AS orders_count,
        RANK() OVER (ORDER BY SUM(o.total_amount) DESC) AS spend_rank
    FROM customers c
    JOIN orders o ON o.customer_id = c.customer_id
    GROUP BY c.customer_id, c.full_name, c.email
)
SELECT customer_id, full_name, email, lifetime_spend, orders_count
FROM ranked
WHERE spend_rank <= 10
ORDER BY lifetime_spend DESC;

-- Top 10 SKUs by revenue AND units sold
CREATE OR REPLACE VIEW v_top_skus AS
WITH sku_stats AS (
    SELECT
        sku,
        category,
        SUM(quantity * unit_price) AS revenue,
        SUM(quantity)              AS units_sold
    FROM order_items
    GROUP BY sku, category
)
SELECT
    sku,
    category,
    revenue,
    units_sold,
    RANK() OVER (ORDER BY revenue    DESC) AS revenue_rank,
    RANK() OVER (ORDER BY units_sold DESC) AS units_rank
FROM sku_stats
WHERE revenue > 0
ORDER BY revenue DESC
LIMIT 10;

-- =========================================================================
-- DATA QUALITY VIEWS (required)
-- =========================================================================

-- DQ 1: Duplicate customers by lowercase email
-- (Uses stored lowercase email per the CHECK constraint)
CREATE OR REPLACE VIEW v_dq_duplicate_emails AS
SELECT
    email,
    COUNT(*)                             AS duplicate_count,
    ARRAY_AGG(customer_id ORDER BY signup_date) AS customer_ids,
    MIN(signup_date)                     AS earliest_signup,
    MAX(signup_date)                     AS latest_signup
FROM customers
GROUP BY email
HAVING COUNT(*) > 1;

-- DQ 2: Orders in clean orders table referencing a customer not in customers
-- NOTE: FK prevents this in the clean table — this view exists so it can
-- still surface *if* the FK is ever disabled / bypassed. Always returns
-- 0 rows under normal operation: a feature, not a bug.
CREATE OR REPLACE VIEW v_dq_orders_missing_customer AS
SELECT o.order_id, o.customer_id, o.order_ts, o.total_amount
FROM orders o
LEFT JOIN customers c ON c.customer_id = o.customer_id
WHERE c.customer_id IS NULL;

-- =========================================================================
-- DATA QUALITY VIEWS (optional stretch)
-- =========================================================================

CREATE OR REPLACE VIEW v_dq_invalid_order_items AS
SELECT * FROM order_items WHERE quantity <= 0 OR unit_price <= 0;

CREATE OR REPLACE VIEW v_dq_invalid_order_status AS
SELECT * FROM orders WHERE status NOT IN ('placed','shipped','cancelled','refunded');

-- =========================================================================
-- QUARANTINE ROLLUP — one-stop view of rejection counts
-- =========================================================================
CREATE OR REPLACE VIEW v_quarantine_summary AS
SELECT 'customers'   AS entity, quarantine_reason, COUNT(*) AS count
FROM quarantine_customers GROUP BY quarantine_reason
UNION ALL
SELECT 'orders',     quarantine_reason, COUNT(*) FROM quarantine_orders GROUP BY quarantine_reason
UNION ALL
SELECT 'order_items',quarantine_reason, COUNT(*) FROM quarantine_order_items GROUP BY quarantine_reason;
```

### 4.3 Config Loading (`src/config.py`)

```python
from pathlib import Path
from pydantic import BaseModel
from pydantic_settings import BaseSettings, SettingsConfigDict
import yaml, os, re

class DBConfig(BaseModel):
    host: str; port: int; database: str; user: str; password: str
    @property
    def dsn(self) -> str:
        return f"postgresql://{self.user}:{self.password}@{self.host}:{self.port}/{self.database}"

class FilesConfig(BaseModel):
    customers: Path; orders: Path; order_items: Path

class ETLConfig(BaseModel):
    batch_size: int = 10000
    allow_null_country_code: bool = True
    default_currency: str = "ZAR"

class LoggingConfig(BaseModel):
    level: str = "INFO"
    format: str

class AgentConfig(BaseModel):
    enabled: bool = True
    provider: str = "groq"
    model: str = "llama-3.3-70b-versatile"

class Config(BaseModel):
    database: DBConfig
    files: FilesConfig
    etl: ETLConfig
    logging: LoggingConfig
    agent: AgentConfig

def load_config(path: str = "config.yaml") -> Config:
    raw = Path(path).read_text()
    # simple ${VAR} interpolation from env
    raw = re.sub(r"\$\{(\w+)\}", lambda m: os.environ.get(m.group(1), ""), raw)
    return Config(**yaml.safe_load(raw))
```

### `src/etl/transform.py`

from typing import Tuple
import pandas as pd

EMAIL_RE = r"^[^@\s]+@[^@\s]+\.[^@\s]+$"
ALLOWED_STATUSES = {"placed", "shipped", "cancelled", "refunded"}

def transform_customers(df: pd.DataFrame) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Returns (clean_df, quarantine_df).

    Steps:
      1. Normalize email (strip, lowercase)
      2. Reject rows with invalid email shape
      3. Parse signup_date → date; reject unparseable
      4. Deduplicate on email: keep earliest signup_date
      5. Cast is_active → bool
    """

def transform_orders(df: pd.DataFrame, valid_customer_ids: set[int]) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Returns (clean_df, quarantine_df).

    Steps:
      1. Parse order_ts with `pd.to_datetime(errors='coerce', utc=True)`
      2. Reject rows with unparseable timestamps
      3. Reject status ∉ ALLOWED_STATUSES
      4. Reject customer_id ∉ valid_customer_ids
      5. Cast total_amount → Decimal; reject negative
    """

def transform_order_items(df: pd.DataFrame, valid_order_ids: set[int]) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Returns (clean_df, quarantine_df).

    Steps:
      1. Reject quantity <= 0
      2. Reject unit_price <= 0
      3. Reject order_id ∉ valid_order_ids
    """

## 6. Load Layer

### Primary path: `COPY` via psycopg 3

```python
# src/etl/load.py (excerpt)
import psycopg
from psycopg import sql

def copy_dataframe(conn: psycopg.Connection, table: str, df: pd.DataFrame) -> int:
    if df.empty:
        return 0
    cols = list(df.columns)
    query = sql.SQL("COPY {} ({}) FROM STDIN WITH (FORMAT CSV)").format(
        sql.Identifier(table),
        sql.SQL(", ").join(map(sql.Identifier, cols)),
    )
    with conn.cursor().copy(query) as cp:
        for row in df.itertuples(index=False, name=None):
            cp.write_row(row)
    return len(df)
```

**Why `COPY` over batched INSERT:**
- 10-100× throughput at scale
- Single network round-trip vs N
- Atomic per-statement; still transactional at the connection level
- Brief explicitly says: *"Prefer client-side COPY"*

**Fallback** (`batched_insert`, commented-out in code) is available if a reviewer runs on a weird psycopg build. Not expected to trigger.

## 7. CLI (`main.py`)

```python
"""
Orders Pipeline CLI.

Usage:
  python main.py init      # create schema + views (idempotent)
  python main.py run       # extract → transform → quality → load
  python main.py report    # generate REPORT.md from views (+ optional agent)
  python main.py truncate  # wipe all tables (dev only; confirms via --yes)
"""
import argparse, sys
from src.config import load_config
from src.db.ddl import apply_schema, apply_views
from src.etl import pipeline
from src.agent.report_agent import generate_report

def main():
    p = argparse.ArgumentParser()
    p.add_argument("command", choices=["init", "run", "report", "truncate"])
    p.add_argument("--config", default="config.yaml")
    p.add_argument("--yes", action="store_true", help="confirm destructive ops")
    args = p.parse_args()

    cfg = load_config(args.config)

    if args.command == "init":
        apply_schema(cfg); apply_views(cfg)
    elif args.command == "run":
        pipeline.run(cfg)
    elif args.command == "report":
        generate_report(cfg)
    elif args.command == "truncate":
        if not args.yes:
            sys.exit("refusing: pass --yes to confirm")
        pipeline.truncate(cfg)

if __name__ == "__main__":
    main()
```

## 11. Error Handling Matrix

| Error | Where caught | Behavior |
|---|---|---|
| File not found | `extract.py` | Raise, exit non-zero |
| Bad email format | `transform_customers` | Quarantine, log |
| Unparseable date | `transform_*` | Quarantine, log |
| Invalid status | `transform_orders` | Quarantine, log |
| Unknown customer FK | `transform_orders` | Quarantine, log |
| Zero/negative qty or price | `transform_order_items` | Quarantine, log |
| Dup email | `transform_customers` | Keep earliest, quarantine loser(s) |
| DB connection refused | `connection.py` | Raise with hint: "is docker compose up?" |
| `COPY` constraint violation | `load.py` | Rollback transaction, log row, raise |
